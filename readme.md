```
program     ->                      { top = null; }
                block
block       ->  '{'                 { saved = top;
                                      top = new Env(top);
                                      print("{ "); }
                decls stmts '}'     { top = saved;
                                      print("} "); }
decls       ->  decls decl
            |   ε
decl        ->  type id ;           { s = new Symbol;
                                      s.type = type.lexeme
                                      top.put(id.lexeme, s); }
stmts       ->  stmts stmt
            |   ε
stmt        ->  block
            |   factor ;            { print("; "); }
factor      ->  id                  { s = top.get(id.lexeme);
                                      print(id.lexeme);
                                      print(":"); }
                                      print(s.type);
```
